4. MA'LUMOTLAR BAZASI (DATABASE) SXEMASI
================================================================================

Eslatma: Sizning eski .db fayllaringiz (tests.db, YANGI_DASTURGA_2024_BAZA.db)
strukturasi o'rganildi. Yangi sxema unga MOS (import qilish oson bo'ladi),
lekin kengaytirilgan.

-- FOYDALANUVCHILAR (Telegram orqali ro'yxatdan o'tganlar)
CREATE TABLE users (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    telegram_id     BIGINT UNIQUE NOT NULL,        -- Telegram user ID
    full_name       TEXT NOT NULL,                 -- foydalanuvchi kiritgan ism
    phone           TEXT NOT NULL,                 -- kontakt orqali yuborilgan
    username        TEXT,                          -- @username (bo'lsa)
    registered_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_active_at  TIMESTAMP,                     -- statistika uchun
    is_blocked      INTEGER DEFAULT 0              -- botni bloklaganlar
);

-- ADMINLAR (admin panel uchun login)
CREATE TABLE admins (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    username        TEXT UNIQUE NOT NULL,
    password_hash   TEXT NOT NULL                  -- pbkdf2:sha256 (eski db kabi)
);

-- MANBALAR (bazalar): "Geometriya", "Arifmetika" ...
CREATE TABLE sources (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    name            TEXT UNIQUE NOT NULL,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- SAVOLLAR
CREATE TABLE source_questions (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    source_id       INTEGER NOT NULL REFERENCES sources(id) ON DELETE CASCADE,
    question_text   TEXT NOT NULL,
    question_type   TEXT NOT NULL DEFAULT 'multiple_choice',
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_q_source ON source_questions(source_id);

-- JAVOB VARIANTLARI
CREATE TABLE source_answers (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    question_id     INTEGER NOT NULL REFERENCES source_questions(id)
                    ON DELETE CASCADE,
    answer_text     TEXT NOT NULL,
    is_correct      INTEGER NOT NULL DEFAULT 0,    -- 0 yoki 1
    position        INTEGER DEFAULT 0              -- tartib (ixtiyoriy)
);
CREATE INDEX idx_a_question ON source_answers(question_id);

MUHIM QOIDA: har bir savolda AYNAN 1 ta to'g'ri javob bo'lishi kerak
(eski bazangizda 100% shunday). Variantlar soni 2 tadan 11 tagacha
bo'lishi mumkin (eski bazada shunday uchraydi) — UI buni hisobga oladi.

-- TESTLAR (foydalanuvchilarga ko'rinadigan)
CREATE TABLE tests (
    id                  INTEGER PRIMARY KEY AUTOINCREMENT,
    name                TEXT UNIQUE NOT NULL,
    time_limit_minutes  INTEGER DEFAULT 0,          -- 0 = cheksiz
    is_active           INTEGER DEFAULT 1,          -- foydalanuvchiga ko'rinadimi
    created_at          TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- TEST QOIDALARI: qaysi manbadan nechta savol
CREATE TABLE test_rules (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    test_id         INTEGER NOT NULL REFERENCES tests(id) ON DELETE CASCADE,
    source_id       INTEGER NOT NULL REFERENCES sources(id) ON DELETE CASCADE,
    question_count  INTEGER NOT NULL                -- nechta savol tushadi
);
-- Misol: test_id=1 uchun 2 ta qator:
--   (test_id=1, source_id=Geometriya, question_count=10)
--   (test_id=1, source_id=Arifmetika, question_count=3)
-- Eski bazadagi "config" JSON o'rniga alohida jadval — qidirish, tahrirlash
-- va validatsiya ancha oson bo'ladi. Eski JSON config import paytida shu
-- jadvalga aylantiriladi.

-- URINISHLAR (foydalanuvchi testni boshlaganda yaratiladi)
CREATE TABLE attempts (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id         INTEGER NOT NULL REFERENCES users(id),
    test_id         INTEGER NOT NULL REFERENCES tests(id),
    started_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    finished_at     TIMESTAMP,                      -- NULL = hali tugamagan
    total_questions INTEGER NOT NULL,
    correct_count   INTEGER DEFAULT 0
);
CREATE INDEX idx_attempt_user ON attempts(user_id);
CREATE INDEX idx_attempt_test ON attempts(test_id);
CREATE INDEX idx_attempt_date ON attempts(started_at);

-- URINISH SAVOLLARI (aynan shu urinishga tushgan random savollar)
CREATE TABLE attempt_questions (
    id                  INTEGER PRIMARY KEY AUTOINCREMENT,
    attempt_id          INTEGER NOT NULL REFERENCES attempts(id)
                        ON DELETE CASCADE,
    question_id         INTEGER NOT NULL REFERENCES source_questions(id),
    order_index         INTEGER NOT NULL,           -- 1, 2, 3...
    shuffled_answers    TEXT,                       -- JSON: aralashtirilgan
                                                    -- javob id'lari tartibi
    selected_answer_id  INTEGER,                    -- NULL = javob berilmagan
    is_correct          INTEGER                     -- NULL = javob berilmagan
);
CREATE INDEX idx_aq_attempt ON attempt_questions(attempt_id);

-- XATOLIK XABARLARI (bug report)
CREATE TABLE error_reports (
    id                  INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id             INTEGER NOT NULL REFERENCES users(id),
    message_text        TEXT,                       -- xatolik mazmuni
    telegram_msg_id     BIGINT,                     -- foydalanuvchi xabari ID
    admin_msg_id        BIGINT,                     -- adminga yuborilgan xabar ID
    status              TEXT DEFAULT 'open',        -- open | fixed
    created_at          TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fixed_at            TIMESTAMP
);

-- BROADCAST TARIXI (ixtiyoriy, lekin foydali)
CREATE TABLE broadcasts (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    admin_tg_id     BIGINT,
    content_type    TEXT,                           -- text/photo/video...
    sent_count      INTEGER DEFAULT 0,
    failed_count    INTEGER DEFAULT 0,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

5. TELEGRAM BOT — TO'LIQ FUNKSIONALLIK VA OQIMLAR
================================================================================

5.1. RO'YXATDAN O'TISH OQIMI (/start)
--------------------------------------
1) Foydalanuvchi /start bosadi.
2) Bot bazani tekshiradi (telegram_id bo'yicha):
   a) AGAR RO'YXATDAN O'TGAN bo'lsa -> to'g'ridan-to'g'ri asosiy menyu.
   b) AGAR YANGI bo'lsa:
      - Bot: "Assalomu alaykum! 👋 Test botiga xush kelibsiz.
              Iltimos, ism-familiyangizni kiriting:"
      - Foydalanuvchi ismini yozadi (FSM holat: waiting_name).
        * Validatsiya: kamida 3 belgi, faqat matn (raqam/emoji emas
          tekshirish shart emas, lekin bo'sh bo'lmasin).
      - Bot: "Rahmat, {ism}! Endi telefon raqamingizni yuboring 📱"
        + REPLY KEYBOARD: [📱 Telefon raqamni yuborish] (request_contact=True)
        (FSM holat: waiting_phone)
      - Foydalanuvchi kontakt yuboradi.
        * MUHIM: contact.user_id == message.from_user.id tekshirilsin —
          boshqa odamning kontaktini yuborib bo'lmasin.
        * Qo'lda raqam yozsa ham qabul qilish mumkin (regex: +998...),
          lekin asosiy yo'l — tugma.
      - Bot bazaga yozadi: telegram_id, full_name, phone, username.
      - Bot: "✅ Ro'yxatdan muvaffaqiyatli o'tdingiz!" + asosiy menyu.

5.4. ⚠️ XATOLIK HAQIDA XABAR BERISH — TO'LIQ OQIM
--------------------------------------
1) Foydalanuvchi "⚠️ Xatolik haqida xabar berish" tugmasini bosadi.
2) Bot (FSM holat: waiting_report):
   "⚠️ Xatolik haqida xabar berish

    Iltimos, testdagi xatolikning SCREENSHOTINI va xatolik MAZMUNINI
    BITTA XABARDA yuboring.

    📌 Masalan: rasm + izoh (rasmga caption yozing).

    Bekor qilish uchun: /cancel"
3) Foydalanuvchi rasm (yoki bir nechta rasm — media group) + caption
   yuboradi. Faqat matn yuborsa ham qabul qilinadi (lekin bot rasm
   bo'lsa yaxshi deb eslatadi — majburiy emas).
4) Bot xabarni ADMINGA FORWARD qiladi va OLDIDAN alohida xabar yuboradi:

   "🚨 XATOLIK ANIQLANDI!

    👤 Foydalanuvchi: Alisher Aliyev
    🆔 ID: 123456789
    📱 Tel: +998901234567
    🕐 Vaqt: 19.07.2026 14:32"

   Keyin foydalanuvchining asl xabari (screenshot + matn) forward
   qilinadi. Media group bo'lsa — hammasi forward qilinadi.

5) Forward qilingan xabar TAGIDA INLINE tugma:
   [ ✅ Xatolik to'g'irlandi ]
   callback_data: "fix_report:{report_id}"

6) Admin xatoni to'g'irlab tugmani bosganda:
   - Bazada report.status = 'fixed', fixed_at = hozirgi vaqt.
   - Tugma matni "✅ To'g'irlandi (19.07.2026)" ga o'zgaradi
     (qayta bosib bo'lmaydi).
   - FOYDALANUVCHIGA avtomatik xabar boradi:
     "✅ Siz yuborgan xatolik to'g'irlandi!
      E'tiboringiz uchun rahmat! 🙏"
7) Foydalanuvchiga: "✅ Xabaringiz adminga yuborildi. Xatolik
   to'g'irlangach sizga xabar beramiz!" — va FSM holatdan chiqadi.

TEXNIK DETALLAR:
  - Media group (bir nechta rasm birga) uchun aiogram'da album
    middleware ishlatiladi (xabarlarni 1 soniya kutib yig'ish).
  - report_id ni callback'ga joylash uchun forward'dan keyin admin
    xabar ID'si bazaga saqlanadi.
  - Bir nechta admin bo'lsa — hammaga yuboriladi, birinchi bosgan
    admin "to'g'irladi" hisoblanadi.

SQL misollar:
  Bugun qo'shilganlar:
    SELECT COUNT(*) FROM users WHERE DATE(registered_at)=DATE('now','localtime');
  Kunlik ishlangan testlar:
    SELECT COUNT(*) FROM attempts
    WHERE finished_at IS NOT NULL
      AND DATE(finished_at)=DATE('now','localtime');
  Eng ko'p ishlangan test:
    SELECT t.name, COUNT(*) c FROM attempts a JOIN tests t ON t.id=a.test_id
    WHERE a.finished_at IS NOT NULL

8. IMPORT TIZIMI — PARSING ALGORITMLARI (ENG MUHIM QISM)
================================================================================

Sizning fayllaringiz tahlili asosida ANIQ qoidalar:

8.1. UMUMIY PRINTSIPLAR (normalizer.py)
--------------------------------------
- Barcha formatlar parse qilingach BIR XIL ichki formatga keladi:
    { "question": str, "answers": [ {"text": str, "correct": bool} ] }
- TOZALASH har bir matnga qo'llanadi:
    • Bosh/oxiridagi bo'shliqlar, ko'p probellar -> bitta probel.
    • Word'dan qolgan **bold** belgilari (agar markdown ko'rinishda
      kelsa) olib tashlanadi.
    • Ko'rinmas belgilar: \u00A0 (nbsp), \u200B (zero-width) olib
      tashlanadi.
    • Savol boshidagi raqamlash olib tashlanadi:
      regex: ^\s*\d+\s*[.\)]\s*  ("1.", "12)", "3 ." ...)
    • Variant boshidagi harf belgilash olib tashlanadi (ixtiyoriy):
      ^[A-DА-Г]\s*[.\)]\s* — LEKIN ehtiyot bo'lish kerak, chunki javob
      "A) ..." bilan boshlanmasligi ham mumkin. Faqat barcha variantlar
      bir xil naqshda bo'lsagina olib tashlash.
- TO'G'RI JAVOB BELGILARI (fayllaringizda uchraydi):
    • "*"  — eng keng tarqalgan (TIF.docx, Тест_МЭД.docx ...)
    • "+"  — ba'zi fayllarda
    • "#"  — zaxira sifatida qo'llab-quvvatlash
  Belgi variant matnining BOSHIDA turadi, lekin:
    • Oldin/keyin probel bo'lishi mumkin: "* Yo'q", "*Yo'q", " * Yo'q"
    • Bold ichida bo'lishi mumkin: "***Матн" (docx'da *** ko'rinib
      qoladi — bir nechta * ketma-ket kelsa hammasi olib tashlanadi)
  Regex: ^\s*[\*\+#]+\s*  -> topilsa correct=True, belgi o'chiriladi.
- VALIDATSIYA har bir savol uchun:
    • question bo'sh emas
    • kamida 2 ta variant
    • aynan 1 ta correct=True (0 ta yoki 2+ bo'lsa -> "muammoli")

8.2. .DOCX PARSER (docx_parser.py)
--------------------------------------
Sizning 4 ta docx faylingizda 2 XIL struktura bor:

FORMAT A — "har savol alohida jadval" (TIF.docx, Тест_МЭД.docx):
  - Hujjatda ko'p jadvallar, har jadval 1 ustunli:
      1-qator: savol matni (raqam bilan boshlanadi)
      keyingi qatorlar: variantlar (to'g'risi * bilan)
  - python-docx: document.tables bo'ylab yurish, har table uchun
    rows[0] -> savol, rows[1:] -> variantlar.

FORMAT B — "bitta katta jadval, savollar aralash"
  (20_ta_Savol-javob..., Qo_mitadan...):
  - BITTA jadvalda hamma narsa qatorlar ketma-ketligida:
      savol qatori (RAQAM bilan boshlanadi: "1.", "**2.**"...),
      keyin uning variantlari, keyin yangi savol qatori...
  - ALGORITM: har qator matnini olamiz;
      agar qator regex ^\s*\**\s*\d+\s*[.\)] ga mos kelsa -> YANGI
      SAVOL boshlandi (oldingi savolni yopamiz);
      aks holda -> joriy savolning varianti.
  - E'TIBOR: docx jadval qatorida "---" separator qatori bo'lishi
    mumkin (markdown eksportda) — bunday qatorlar tashlab yuboriladi.

UNIVERSAL DOCX ALGORITM (ikkala formatni ham yeydi):
  1) Hujjatdagi BARCHA jadval qatorlarini tartib bilan tekis
     ro'yxatga yig'amiz (har qator = bitta matn; ko'p ustunli bo'lsa
     ustunlar birlashtiriladi).
  2) Jadvaldan tashqaridagi paragraflar (sarlavhalar: "ТЕСТ САВОЛ
     ЖАВОБЛАРИ" kabi) E'TIBORGA OLINMAYDI (lekin agar hujjatda umuman
     jadval bo'lmasa — paragraflar ro'yxati olinadi va TXT algoritmi
     qo'llanadi, chunki ba'zi docx'lar jadvalsiz bo'lishi mumkin).
  3) Tekis ro'yxat ustida "savol / variant" ajratish:
     - Qator savolmi? -> raqam naqshi bilan boshlansa: ^\**\s*\d+\s*[.\)]
     - Yoki (FORMAT A uchun): yangi table boshlanishi ham "yangi savol"
       signali — parser table chegaralarini biladi.
  4) Har variant qatorida to'g'ri javob belgisi (* + #) tekshiriladi.
  5) Normalizatsiya + validatsiya.
  6) Bo'sh qatorlar, faqat "---" bo'lgan qatorlar tashlanadi.
  7) Takror savollar (fayl ichida ham) aniqlanadi (TIF.docx'da 5- va
     6-savol bir xil edi!) — preview'da "fayl ichida takror" deb
     ko'rsatiladi.

8.3. .TXT PARSER (txt_parser.py)
--------------------------------------
TXT fayllarda keng tarqalgan formatlar (hammasi qo'llab-quvvatlanadi):

  FORMAT 1 (bo'sh qator bilan ajratilgan bloklar):
      1. Savol matni?
      A) variant
      *B) to'g'ri variant
      C) variant

      2. Keyingi savol...

  FORMAT 2 (+ belgili):
      Savol matni?
      +to'g'ri javob
      noto'g'ri javob
      noto'g'ri javob

  ALGORITM:
  1) Fayl UTF-8 da o'qiladi (BOM bilan ham: encoding='utf-8-sig';
     agar UnicodeDecodeError -> cp1251 bilan urinish — eski kirill
     fayllar uchun).
  2) Qatorlarga bo'linadi, bo'sh qatorlar blok ajratuvchi hisoblanadi.
  3) Agar bloklar aniq bo'lsa: blokning 1-qatori savol, qolganlari
     variantlar.
  4) Agar bo'sh qatorlar tartibsiz bo'lsa: raqam naqshi bo'yicha
     ajratish (docx'dagi kabi).
  5) To'g'ri javob belgisi va normalizatsiya — umumiy qoidalar.

  ALGORITM:

  2) Jadval mavjudligi tekshiriladi (sqlite_master). Agar struktura

  3) IKKI XIL IMPORT REJIMI (admin tanlaydi):

16. XAVFSIZLIK
================================================================================

  1) Telegram initData HMAC tekshiruvi — MAJBURIY (soxta so'rovlardan
     himoya). Kod: secret_key = HMAC_SHA256("WebAppData", bot_token);
     hash = HMAC_SHA256(secret_key, data_check_string) taqqoslanadi.
  2) To'g'ri javoblar frontendga faqat javob berilgandan KEYIN
     yuboriladi (savol yuklashda emas).
  3) Admin parollari pbkdf2:sha256 (yoki bcrypt) bilan hash.
  4) JWT: qisqa muddat, SECRET_KEY .env da, hech qachon git'ga
     yozilmaydi.
  5) SQL injection: faqat ORM/parametrlangan so'rovlar.
  6) XSS: barcha foydalanuvchi/savol matnlari render'dan oldin
     escape qilinadi (innerHTML o'rniga textContent; HTML kerak
     bo'lsa faqat oq ro'yxatdagi teglar).
  7) Yuklangan fayllar: kengaytma + hajm tekshiruvi, vaqtinchalik
     papkada, ishlov berilgach o'chiriladi. .db read-only ochiladi.
  8) Admin API'lari alohida JWT scope bilan himoyalangan.
  9) HTTPS majburiy (Telegram Mini App http'da ishlamaydi).
  10) attempt'ga faqat EGASI kirishi tekshiriladi
      (attempt.user_id == token.user_id).
